 * @returns {ArraySSI}
 */
function createBatchSSI(data, domain) {
    console.log("New BATCH_SSI in domain", domain);
    const openDSU = require('opendsu');
    const keyssiSpace = openDSU.loadApi("keyssi");
    let hint;
    if (data[openDSU.constants.BRICKS_DOMAIN_KEY]) {
        hint = {};

    }
